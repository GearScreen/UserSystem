generator client {
  provider = "prisma-client-js"
  // output   = "../app/generated/prisma"
}

model User {
  id        Int      @id @default(autoincrement())
  email     String   @unique
  username        String?   @unique @db.VarChar(100)
  role            Role      @default(USER)
  status          Status    @default(ACTIVE)

  // Password
  passwordHash    String    @db.VarChar(255) @map("password_hash")
  resetToken      String?   @unique @map("password_reset_token")
  resetExpires    DateTime? @map("password_reset_expires_at")

  // Security
  lockedUntil     DateTime? @map("locked_until")

  // Account Verification
  emailVerified  Boolean @default(false) @map("email_verified")
  verificationToken     String?   @unique @map("verification_token")
  verificationExpiresAt DateTime? @map("verification_expires_at")

  // Timestamps
  createdAt       DateTime  @default(now()) @map("created_at")
  updatedAt       DateTime  @updatedAt @map("updated_at")

  @@index([email])
  @@index([role])
  @@map("users")
}
